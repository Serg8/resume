                function addClass(obj, cls) {
                  let flag = false;
                  let arr = obj.className.split(' ');
                  for(let i = 0; i < arr.length; i++) {
                    if(arr[i] == cls){
                        flag = true;
                        break;
                      }
                  }
                  if(!flag) {
                    arr.push(cls);
                    obj.className = arr.join(" ").trim();
                  }
                  return obj;
                }
